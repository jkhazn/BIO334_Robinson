def find_peaks(graph):
    peak_dic = {}
    for v in graph.vs:
        index_list = graph.neighborhood(v, mode='out', order=1, mindist=1)
        source_fitness = v['Fitness']
        is_peak = not any(graph.vs[i]['Fitness'] >= source_fitness for i in index_list)
        if is_peak:
            peak_dic[v['Sequence']] = source_fitness
    return peak_dic

peak_dic = find_peaks(graph)

Random Walk Simulation

Greedy uphill random walk from low-fitness starting points; repeated 1000 times. Set eval=TRUE to render (takes ~60 s and prints one line per walk).

F10 = min(list_of_fitness) + (max(list_of_fitness) - min(list_of_fitness)) * 0.1
lowest_fitnesses = [seq for seq, fit in dictionary.items() if fit < F10]

#random.seed(10)
lowest_fitnesses_random = random.sample(lowest_fitnesses, 1).pop()

starting_point = graph.vs.find(Sequence=lowest_fitnesses_random).index

if graph.vs[starting_point]['Sequence'] in peak_dic:
    print('Dont continue, the randmo walk would start at a point')

stepcount = []
for i in range(1000):
    lowest_fitnesses_random = random.sample(lowest_fitnesses, 1).pop()

    starting_point = graph.vs.find(Sequence=lowest_fitnesses_random).index

    steps = 0
    while True:
        neighbour_list = graph.neighborhood(starting_point, mode='out', order=1, mindist=1)
        g_mut = random.sample(neighbour_list, k=1).pop()

        fitness_start = graph.vs[starting_point]['Fitness']
        fitness_end = graph.vs[g_mut]['Fitness']

        if fitness_end > fitness_start:
            starting_point = g_mut
            steps += 1
            if graph.vs[starting_point]['Sequence'] in peak_dic:
                break
    stepcount.append(steps)
    print(graph.vs[starting_point]['Sequence'], graph.vs[starting_point]['Fitness'], steps)

print(stepcount)

plt.hist(stepcount, bins=50, alpha=0.5, label='Peaks', color='red')

plt.title('Pairwise Hamming distance: peaks vs random sample')
