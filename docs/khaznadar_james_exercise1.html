fs <- list.files(here("data", "files_for_ex1"), ".count$", full.names = TRUE)



my_function <- function(filepath) {
  file <- read.table(filepath)
  sum_counts <- sum(file[, 2])
  CPM = 1e6 * (file[,2] / sum_counts) 
  CPM_greater_1 = sum(CPM > 1)
  return(CPM_greater_1)
}
